#include<iostream>
#include<set>
#include<vector>
#include<algorithm>

int main()
{
	int T,N;
	std::pair<int, int> pairs;
	std::set<std::pair<int,int>> s;

	std::cin >> T;
	std::cin >> N;
	
	int *number=new int[N];
	for (int i = 0; i < N; i++)
	{
		std::cin >> number[i];
	}

	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < N; j++)
		{
			if (i != j)
			{
				std::cout << " (" << number[i] << "," << number[j] << ")" << "   ";
				std::pair<int, int> x = std::make_pair(number[i], number[j]);
				s.insert(x);
				
			}
		}
	}

	std::cout << "\n";

	std::vector<int> v;
	for (auto e : s)
	{
		int x = e.first;
		int y = e.second;
		int count = 0;
		for (int i = 1; i < x; i++)
		{
			for (int j = 1; j < y; j++)
			{
				count++;
			}
		}
		v.push_back(count);
	}
	 int k=*max_element(v.begin(),v.end());
	 std::cout << k;

	return 0;
}
